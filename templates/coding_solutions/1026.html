Given the root of a binary tree, find the maximum value V for which there exists different nodes A and B where V = |A.val - B.val| and A is an ancestor of B.

(A node A is an ancestor of B if either: any child of A is equal to B, or any child of A is an ancestor of B.)

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: void diff(TreeNode* node, int minnum, int maxnum, int &maxdiff) { if(node == nullptr) return; minnum = min(node->val, minnum); maxnum = max(node->val, maxnum); maxdiff = max(abs(minnum - maxnum), maxdiff); diff(node->left, minnum, maxnum, maxdiff); diff(node->right, minnum, maxnum, maxdiff); } int maxAncestorDiff(TreeNode* root) { int maxDiff = INT_MIN; diff(root, INT_MAX, INT_MIN, maxDiff); return maxDiff; } }; {% endblock solution_code %}

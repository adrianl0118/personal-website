Given a binary tree, find its maximum depth.

The maximum depth is the number of nodes along the longest path from the root node down to the farthest leaf node.

Note: A leaf is a node with no children.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int maxDepth(TreeNode* root) { int max = 0; findMaxDepth(root,0,max); return max; } void findMaxDepth(TreeNode* root, int depth, int& max){ if (!root){ return; } else if (!root->left && !root->right){ depth++; if (depth > max){ max = depth; } return; } depth++; if (root->left){ findMaxDepth(root->left,depth,max); } if (root->right){ findMaxDepth(root->right,depth,max); } return; } }; {% endblock solution_code %}

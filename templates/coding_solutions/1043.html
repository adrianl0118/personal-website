Given an integer array A, you partition the array into (contiguous) subarrays of length at most K. After partitioning, each subarray has their values changed to become the maximum value of that subarray.

Return the largest sum of the given array after partitioning.

{% endblock question %} {% block solution_exp %}

Memoization solution.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int maxSumAfterPartitioning(vector& A, int K) { vector memo(A.size()+1,-1); return maxSum(memo,A,K,0); } int maxSum(vector& memo, vector& A, int& K, int pos){ if (pos >= A.size())return 0; if (memo[pos] != -1)return memo[pos]; int range_max = A[pos], rightsum = 0; //For all possible sizes of subarray extending right from this point, get sum of the array plus //cumulative sum of values to the right of the array's right bound. for (int i = pos; i < A.size() && i < pos+K; i++){ if (A[i] > range_max){ range_max = A[i]; } rightsum = max(rightsum,range_max*(i-pos+1)+maxSum(memo,A,K,i+1)); //keep track of max observed sum } return memo[pos] = rightsum; } }; {% endblock solution_code %}

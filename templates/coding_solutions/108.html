Given an array where elements are sorted in ascending order, convert it to a height balanced BST.

For this problem, a height-balanced binary tree is defined as a binary tree in which the depth of the two subtrees of every node never differ by more than 1.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: TreeNode* sortedArrayToBST(vector& nums) { return buildTree(nums,0,nums.size()-1); } TreeNode* buildTree(vector& nums, int start, int end){ if (start > end) return NULL; int mid = start + (end-start)/2; TreeNode* root = new TreeNode(nums[mid]); root->left = buildTree(nums,start,mid-1); root->right = buildTree(nums,mid+1,end); return root; } }; {% endblock solution_code %}

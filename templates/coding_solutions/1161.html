Given the root of a binary tree, the level of its root is 1, the level of its children is 2, and so on.

Return the smallest level X such that the sum of all the values of nodes at level X is maximal.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int maxLevelSum(TreeNode* root) { unordered_map levelsum; getLevelSums(root, 1, levelsum); int themax = 0; int mlevel = INT_MAX; for (unordered_map::iterator it = levelsum.begin(); it != levelsum.end(); it++){ if (it->second > themax){ mlevel = it->first; themax = it->second; } } return mlevel; } void getLevelSums(TreeNode* root, int level, unordered_map& levelsum){ if (!root){return;} levelsum[level]+=root->val; getLevelSums(root->left,level+1,levelsum); getLevelSums(root->right,level+1,levelsum); return; } }; {% endblock solution_code %}

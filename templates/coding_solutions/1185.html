Given a date, return the corresponding day of the week for that date.

The input is given as three integers representing the day, month and year respectively.

Return the answer as one of the following values {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"}.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { vector<string> dows {"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"}; vector<int> monthDays {0, 3, 0, 3, 2, 3, 2, 3, 3, 2, 3, 2, 3}; public: string dayOfTheWeek(int day, int month, int year) { // 1971/01/01: Friday int m = 3; // year m += (year - 1971) + (year - 1969) / 4; // month for(int i = 1; i < month; i++) m += monthDays[i]; if(year % 4 == 0 && month > 2) ++m; // day m += day; // final answer return dows[m % 7]; } }; {% endblock solution_code %}

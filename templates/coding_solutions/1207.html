Given an array of integers arr, write a function that returns true if and only if the number of occurrences of each value in the array is unique.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: bool uniqueOccurrences(vector& arr) { unordered_map numcount; for (auto& n : arr){ numcount[n]++; } set uniquecounts; for (unordered_map::iterator it = numcount.begin(); it != numcount.end(); it++){ uniquecounts.insert(it->second); } return (uniquecounts.size() == numcount.size()); } }; {% endblock solution_code %}

On an 8x8 chessboard, there can be multiple Black Queens and one White King.

Given an array of integer coordinates queens that represents the positions of the Black Queens, and a pair of coordinates king that represent the position of the White King, return the coordinates of all the queens (in any order) that can attack the King.

{% endblock question %} {% block solution_exp %}

Iterate over all queens and determine if any of them are in viable positions of attack relative to the King's position. The yes vector accumulates attackers from all positions and is returned.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector> queensAttacktheKing(vector>& queens, vector& king) { vector> yes; vector top, bottom, left, right, topleft, topright, bottomleft, bottomright; for (auto q : queens){ if (q[0] == king[0]) { if (q[1] < king[1]){ if (left.empty() || left[1] < q[1]){ left = q; } } else { if (right.empty() || right[1] > q[1]){ right = q; } } } else if (q[1] == king[1]) { if (q[0] < king[0]){ if (top.empty() || top[0] < q[0]){ top = q; } } else { if (bottom.empty() || bottom[0] > q[0]){ bottom = q; } } } else if (abs(q[0]-king[0]) == abs(q[1]-king[1])) { if (q[0] < king[0] && q[1] < king[1]){ if (topleft.empty() || (topleft[0] < q[0] && topleft[1] < q[1])){ topleft = q; } } else if (q[0] > king[0] && q[1] < king[1]) { if (bottomleft.empty() || (bottomleft[0] > q[0] && bottomleft[1] < q[1])){ bottomleft = q; } } else if (q[0] < king[0] && q[1] > king[1]) { if (topright.empty() || (topright[0] < q[0] && topright[1] > q[1])){ topright = q; } } else if (q[0] > king[0] && q[1] > king[1]){ if (bottomright.empty() || (bottomright[0] > q[0] && bottomright[1] > q[1])){ bottomright = q; } } } } if (!top.empty()){yes.push_back(top);} if (!bottom.empty()){yes.push_back(bottom);} if (!left.empty()){yes.push_back(left);} if (!right.empty()){yes.push_back(right);} if (!topleft.empty()){yes.push_back(topleft);} if (!bottomleft.empty()){yes.push_back(bottomleft);} if (!topright.empty()){yes.push_back(topright);} if (!bottomright.empty()){yes.push_back(bottomright);} return yes; } }; {% endblock solution_code %}

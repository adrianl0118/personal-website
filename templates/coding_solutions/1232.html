You are given an array coordinates, coordinates[i] = [x, y], where [x, y] represents the coordinate of a point. Check if these points make a straight line in the XY plane.

{% endblock question %} {% block solution_exp %}

Confirm uniformity of slope between all adjacent points.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: bool checkStraightLine(vector<vector<int>>& coordinates) { double ratio = (double(coordinates[1][1])-double(coordinates[0][1]))/(double(coordinates[1][0])-double(coordinates[0][0])); for (int i = 2; i < coordinates.size(); i++){ double cur = (double(coordinates[i][1])-double(coordinates[i-1][1]))/(double(coordinates[i][0])-double(coordinates[i-1][0])); if (cur != ratio) return false; } return true; } }; {% endblock solution_code %}

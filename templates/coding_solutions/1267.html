You are given a map of a server center, represented as a m * n integer matrix grid, where 1 means that on that cell there is a server and 0 means that it is no server. Two servers are said to communicate if they are on the same row or on the same column.

Return the number of servers that communicate with any other server.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int countServers(vector<vector<int>>& grid) { int count = 0; for (int i = 0; i < grid.size(); i++){ for (int j = 0; j < grid[0].size(); j++){ if (grid[i][j] == 1){ bool comm = false; for (int k = 0; k < grid.size(); k++){ if (k == i) continue; if (grid[k][j] == 1){ count++; comm = true; break; } } if (!comm){ for (int l = 0; l < grid[0].size(); l++){ if (l == j) continue; if (grid[i][l] == 1){ count++; break; } } } } } } return count; } }; {% endblock solution_code %}

Given a binary tree, return the sum of values of its deepest leaves.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int deepestLeavesSum(TreeNode* root) { int maxDepth = 0; int sum = 0; getMaxDepth(root, 0, maxDepth); getDeepestLeaves(root, 0, maxDepth, sum); return sum; } void getMaxDepth(TreeNode* root, int depth, int& maxDepth){ if (!root){ return; } if (depth > maxDepth){ maxDepth = depth; } if (root->left){ getMaxDepth(root->left, depth+1, maxDepth); } if (root->right){ getMaxDepth(root->right, depth+1, maxDepth); } } void getDeepestLeaves(TreeNode* root, int depth, int maxDepth, int& sum){ if (!root){ return; } if (depth == maxDepth){ sum += root->val; } getDeepestLeaves(root->left, depth+1, maxDepth, sum); getDeepestLeaves(root->right, depth+1, maxDepth, sum); return; } }; {% endblock solution_code %}

Given an integer n, return any array containing n unique integers such that they add up to 0.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector sumZero(int n) { vector ans; int curr = 1; while (ans.size() < n){ if (n-ans.size() > 1){ ans.push_back(curr); ans.push_back(-1*curr); curr++; } else { ans.push_back(0); } } return ans; } }; {% endblock solution_code %}

Given two binary search trees root1 and root2.

Return a list containing all the integers from both trees sorted in ascending order.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector getAllElements(TreeNode* root1, TreeNode* root2) { vector ans; DFS(root1, ans); DFS(root2, ans); sort(begin(ans),end(ans)); return ans; } void DFS(TreeNode* root, vector& ans){ if (!root) return; ans.push_back(root->val); DFS(root->left, ans); DFS(root->right,ans); return; } }; {% endblock solution_code %}

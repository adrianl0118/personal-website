Given a m * n matrix mat and an integer K, return a matrix answer where each answer[i][j] is the sum of all elements mat[r][c] for i - K <= r <= i + K, j - K <= c <= j + K, and (r, c) is a valid position in the matrix.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector> matrixBlockSum(vector>& mat, int K) { vector> ans; for (int i = 0; i < mat.size(); i++){ vector row; int minx, maxx, miny, maxy; if (i - K < 0){ minx = 0; } else { minx = i - K;} if (i + K >= mat.size()){ maxx = mat.size()-1; } else { maxx = i + K; } for (int j = 0; j < mat[0].size(); j++){ if (j - K < 0){ miny = 0; } else { miny = j - K;} if (j + K >= mat[0].size()){ maxy = mat[0].size()-1; } else { maxy = j + K; } int sum = 0; for (int x = minx; x <= maxx; x++){ for (int y = miny; y <= maxy; y++){ sum += mat[x][y]; } } row.push_back(sum); } ans.push_back(row); } return ans; } }; {% endblock solution_code %}

Given a positive integer num consisting only of digits 6 and 9.

Return the maximum number you can get by changing at most one digit (6 becomes 9, and 9 becomes 6).

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int maximum69Number (int num) { string number = to_string(num); for (int i = 0; i < number.size(); i++){ if (number[i] == '6'){ number[i] = '9'; break; } } return std::stoi(number); } }; {% endblock solution_code %}

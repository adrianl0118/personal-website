Given an array of integers arr, replace each element with its rank.

The rank represents how large the element is. The rank has the following rules:

  • Rank is an integer starting from 1.
  • The larger the element, the larger the rank. If two elements are equal, their rank must be the same.
  • Rank should be as small as possible.
{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector<int> arrayRankTransform(vector<int>& arr) { //Get array in sorted order vector<int> dummy = arr; sort(dummy.begin(),dummy.end()); //Store number, rank without duplicates in a map int i = 1; unordered_map <int,int> NumRank; for(auto& d : dummy) if(NumRank.find(d) == NumRank.end()) NumRank[d] = i++; //Populate vector with ranks of numbers inserted with reference to original ordering of numbers i = 0; for(auto a : arr) dummy[i++] = NumRank[a]; return dummy; } }; {% endblock solution_code %}

Given a m * n matrix mat of ones (representing soldiers) and zeros (representing civilians), return the indexes of the k weakest rows in the matrix ordered from the weakest to the strongest.

A row i is weaker than row j, if the number of soldiers in row i is less than the number of soldiers in row j, or they have the same number of soldiers but i is less than j. Soldiers are always stand in the frontier of a row, that is, always ones may appear first and then zeros.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector kWeakestRows(vector>& mat, int k) { //Row index + count of 1s vector RowSoldier; //Keep track of smallest numbers and avoid duplicates priority_queue, greater> pq; set prevcounts; //Count all soldiers in each row, avoiding duplicate counts when inserting to priority queue for (int i = 0; i < mat.size(); i++){ int j; for (j = 0; j < mat[i].size(); j++){ if (mat[i][j] == 0){ break; } } RowSoldier.push_back(j); if (prevcounts.find(j) == prevcounts.end()){ pq.push(j); prevcounts.insert(j); } } //Retrieve k weakest vector ans; int answercount = 0; while (!pq.empty() && answercount < k){ int count = pq.top(); pq.pop(); for (int l = 0; l < RowSoldier.size(); l++){ if (RowSoldier[l] == count){ ans.push_back(l); answercount++; if (answercount >= k){ break; } } } } return ans; } }; {% endblock solution_code %}

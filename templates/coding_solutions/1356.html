Given an integer array arr. You have to sort the integers in the array in ascending order by the number of 1's in their binary representation and in case of two or more integers have the same number of 1's you have to sort them in ascending order.

Return the sorted array.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector sortByBits(vector& arr) { //Get all the onebit counts of all numbers and store together vector> bitnum; for (auto& a : arr){ bitnum.push_back(make_pair(getOneBits(a),a)); } //lambda sort in ascending order by bitcount, if bitcount the same sort in ascending order by number value sort(bitnum.begin(), bitnum.end(), [](auto& p1, auto& p2){ return p1.first < p2.first || (p1.first == p2.first && p1.second < p2.second); }); //populate the answer vector with the original numbers in the sorted order and return vector ans; for (auto& b : bitnum){ ans.push_back(b.second); } return ans; } int getOneBits(int num){ unsigned int count = 0; while (num) { count += num & 1; num >>= 1; } return count; } }; {% endblock solution_code %}

Given the array nums, for each nums[i] find out how many numbers in the array are smaller than it. That is, for each nums[i] you have to count the number of valid j's such that j != i and nums[j] < nums[i].

Return the answer in an array.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector smallerNumbersThanCurrent(vector& nums) { vector ans; for (int i = 0; i < nums.size(); i++){ int count = 0; for (int j = 0; j < nums.size(); j++){ if (j != i && nums[j] < nums[i]){ count++; } } ans.push_back(count); } return ans; } }; {% endblock solution_code %}

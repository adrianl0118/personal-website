Given a non-empty array of integers, every element appears three times except for one, which appears exactly once. Find that single one.

Your algorithm should have a linear runtime complexity. Could you implement it without using extra memory?

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int singleNumber(vector<int>& nums) { int n = nums.size(); long ans = 0; //For all bits in a standard 32-bit integer for(int i = 0; i < 32; i++){ int onebits = 0; //calculate sum of 1 bits in this position for all words for(int j = 0; j < n; j++){ onebits += nums[j]&1; //adds 1 if nums[j] has a one bit nums[j] >>= 1; //values in nums are right-shifted so that at each i iteration, the smallest bit is the bit in question } //If the one bits are not a multiple of 3, the unique number has a one bit at this position, add to answer ans += (onebits % 3)*pow(2,i); } return ans; } }; {% endblock solution_code %}

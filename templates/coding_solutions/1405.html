A string is called happy if it does not have any of the strings 'aaa', 'bbb' or 'ccc' as a substring.

Given three integers a, b and c, return any string s, which satisfies following conditions:

  • s is happy and longest possible.
  • s contains at most a occurrences of the letter 'a', at most b occurrences of the letter 'b' and at most c occurrences of the letter 'c'.
  • s will only contain 'a', 'b' and 'c' letters.

If there is no such string s return the empty string "".

{% endblock question %} {% block solution_exp %}

As we iteratively build the string we always want to add two of the greatest remaining frequency character if possible, with a single instance of the second-highest frequency character added immediately to space out a string of pairs of the most abundant character.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: string longestDiverseString(int a, int b, int c) { string ans = ""; priority_queue> countChar; //Track non-zero counts of available letters from greatest to least if (a) countChar.push(make_pair(a,'a')); if (b) countChar.push(make_pair(b,'b')); if (c) countChar.push(make_pair(c,'c')); while (!countChar.empty()) { //Iteratively build the string. At any state, add the letters that are not the last letter //Add two of the maximum frequency letter if possible, else 1 int count1 = countChar.top().first; char ch1 = countChar.top().second; countChar.pop(); int add = (!ans.empty() && ans.back() == ch1) ? 1 : 2; //add 2 if the last letter is not this letter, else one for (int i = 0; i < add && count1 > 0; i++) { ans.push_back(ch1); count1--; } if (countChar.empty()) break; //if there are no different characters left to add, then string is complete //Add one of the second most-frequent character to space out groups of the most frequent character int count2 = countChar.top().first; char ch2 = countChar.top().second; countChar.pop(); ans.push_back(ch2); count2--; if (count1) countChar.push(make_pair(count1,ch1)); //add the two used lettercounts back to the priority queue if greater than 0 if (count2) countChar.push(make_pair(count2,ch2)); } return ans; } }; {% endblock solution_code %}

Given an array of string words. Return all strings in words which is substring of another word in any order.

String words[i] is substring of words[j], if can be obtained removing some characters to left and/or right side of words[j].

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector<string> stringMatching(vector<string>& words) { vector<string> ans; for (int i = 0; i < words.size(); i++){ for (int j = 0; j < words.size(); j++){ if (j != i && words[j].find(words[i]) != -1){ ans.push_back(words[i]); break; } } } return ans; } }; {% endblock solution_code %}

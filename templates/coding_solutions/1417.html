Given alphanumeric string s. (Alphanumeric string is a string consisting of lowercase English letters and digits).

You have to find a permutation of the string where no letter is followed by another letter and no digit is followed by another digit. That is, no two adjacent characters have the same type.

Return the reformatted string or return an empty string if it is impossible to reformat the string.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: string reformat(string s) { //Separate digits and letters string digi = ""; string lett = ""; for (auto& c : s){ if (isdigit(c)) digi+=c; else lett += c; } //If difference in size of digi and letters exceeds 1, an alternating sequence is impossible int a = digi.size(); int b = lett.size(); if (fabs(a - b) > 1) return ""; //Build the answer string string ans = ""; if (digi.size() < lett.size()) swap(digi,lett); //ensure digi is the larger string for (int i = 0; i < lett.size(); i++){ //build section by section: longer string letter, shorter string letter ans += digi[i]; ans += lett[i]; } if (digi.size() > lett.size()) ans += digi[lett.size()]; //add remaining letter in longer string if there is one and return answer return ans; } }; {% endblock solution_code %}

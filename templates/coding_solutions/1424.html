Given a 2D matrix of integers, nums, return all elements of nums in diagonal order from bottom-left to upper right over all diagonals in the matrix.

Rows in nums may be of different sizes.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector<int> findDiagonalOrder(vector<vector<int>>& nums) { //Create vector for grouping numbers by diagonal (for any diagonal, i+j is constant) vector<vector<int>> buckets; vector<int> ans; //Iterating over the matrix for (int i = 0; i < nums.size(); i++) for (int j = 0; j < nums[i].size(); j++) if (buckets.size() <= i + j) //Create a new diagonal if needed buckets.push_back({nums[i][j]}); else //Use an existing diagonal buckets[i + j].push_back(nums[i][j]); //Store the numbers in order for (int i = 0; i < buckets.size(); i++) for (int j = buckets[i].size() - 1; j >= 0; j--) //Read answers in each diagonal in reverse because the 2D traversal stores them in upper-right to bottom-left order and we need the opposite ans.push_back(buckets[i][j]); return ans; } }; {% endblock solution_code %}

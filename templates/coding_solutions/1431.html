Given the array candies and the integer extraCandies, where candies[i] represents the number of candies that the ith kid has.

For each kid check if there is a way to distribute extraCandies among the kids such that he or she can have the greatest number of candies among them. Notice that multiple kids can have the greatest number of candies.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector<bool> kidsWithCandies(vector<int>& candies, int extraCandies) { int g = *max_element(begin(candies),end(candies)); vector<bool> ans; for (auto& c : candies) if (c + extraCandies < g) ans.push_back(false); else ans.push_back(true); return ans; } }; {% endblock solution_code %}

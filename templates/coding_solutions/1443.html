Given an undirected tree consisting of n vertices numbered from 0 to n-1, which has some apples in their vertices. You spend 1 second to walk over one edge of the tree. Return the minimum time in seconds you have to spend in order to collect all apples in the tree starting at vertex 0 and coming back to this vertex.

The edges of the undirected tree are given in the array edges, where edges[i] = [fromi, toi] means that exists an edge connecting the vertices fromi and toi. Additionally, there is a boolean array hasApple, where hasApple[i] = true means that vertex i has an apple, otherwise, it does not have any apple.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int minTime(int n, vector<vector<int>>& edges, vector<bool>& hasApple) { if (find(begin(hasApple),end(hasApple),true) == hasApple.end()) return 0; //Hash all destinations to origins for ease of access unordered_map<vector<int>> fromTos; for (auto& e : edges) fromTos[e[0]].push_back(e[1]); //Do the DFS return branchTime(0,fromTos,hasApple); } int branchTime(int start, unordered_map<vector<int>>& fromTos, vector<bool>& hasApple){ int time = 0; for (auto& t : fromTos[start]){ int BT = branchTime(t,fromTos,hasApple); if (BT != 0 || hasApple[t]) //Add the time to traverse to and from the current node if it has an apple or if there is an apple further down one of its branches time += (BT+2); } return time; } }; {% endblock solution_code %}

Given a column title as appear in an Excel sheet, return its corresponding column number.

For example:

A -> 1
B -> 2
C -> 3
...
Z -> 26
AA -> 27
AB -> 28
...

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int titleToNumber(string s) { if(s.empty()) return 0; int i = 0; long ans = 0; while (i < s.size()){ ans = ans*26+s[i]-'A'+1; i++; } return ans; } }; {% endblock solution_code %}

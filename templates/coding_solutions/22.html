Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses.

For example, given n = 3, a solution set is:

"((()))",
"(()())",
"(())()",
"()(())",
"()()()"

{% endblock question %} {% block solution_exp %}

Recursively generate all possible permutations of valid bracket sequences taking into account open brackets and brackets that are unused, DFS to the leaves where we push the sequences to the answer vector.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector generateParenthesis(int n) { vector perms; string curr = "("; int open = n-1; int closed = n; int isOpen = 1; generatePerms(curr, perms, open, closed, isOpen); return perms; } void generatePerms(string curr, vector& perms, int open, int closed, int isOpen){ if (open == 0 && closed == 0){ perms.push_back(curr); return; } if (isOpen && closed > 0){ generatePerms(curr+')',perms,open,closed-1,isOpen-1); } if (open > 0){ generatePerms(curr+'(',perms,open-1,closed,isOpen+1); } return; } }; {% endblock solution_code %}

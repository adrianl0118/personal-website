Given two arrays, write a function to compute their intersection.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector intersection(vector& nums1, vector& nums2) { //remove duplicates from nums1 by changing to set, clear nums1 and re-use as storage set s( nums1.begin(), nums1.end() ); nums1.clear(); //populate nums1 with elements that are in the set as well as nums2, avoiding duplicates for (auto& n : nums2){ if (s.count(n) && find(nums1.begin(),nums1.end(),n) == nums1.end()){ nums1.push_back(n); } } return nums1; } }; {% endblock solution_code %}

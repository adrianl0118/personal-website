Given an n-ary tree, return the level order traversal of its nodes' values.

Nary-Tree input serialization is represented in their level order traversal, each group of children is separated by the null value (See examples).

{% endblock question %} {% block solution_exp %}

Two solutions; DFS and BFS as shown below.

{% endblock solution_exp %} {% block solution_code %} /* DFS */ class Solution { public: vector> levelOrder(Node* root) { vector> ans; DFS(root,ans, 0); return ans; } void DFS(Node* root, vector>& ans, int depth){ if (!root) return; if (ans.empty() || ans.size() <= depth) ans.push_back({}); ans[depth].push_back(root->val); for (auto& c : root->children){ DFS(c,ans,depth+1); } return; } }; /* BFS */ class Solution { public: vector> levelOrder(Node* root) { vector> ans; if (!root) return {}; queue BFS; BFS.push(root); while (!BFS.empty()){ int levelsize = BFS.size(); vector level; for (int i = 0; i < levelsize; i++){ Node* curr = BFS.front(); BFS.pop(); for (auto& c : curr->children){ BFS.push(c); } level.push_back(curr->val); } ans.push_back(level); } return ans; } }; {% endblock solution_code %}

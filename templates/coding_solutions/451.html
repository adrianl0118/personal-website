Given a string, sort it in decreasing order based on the frequency of characters.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: string frequencySort(string s) { //Count map CharCount; for (char c: s){ CharCount[c]++; } //Remove count duplicates and sort set Counts; for (map::iterator it = CharCount.begin(); it != CharCount.end(); it++){ Counts.insert(it->second); } priority_queue OrderedCounts; for (auto a : Counts){ OrderedCounts.push(a); } //Write and return answer string ans = ""; while (!OrderedCounts.empty()){ int currCount = OrderedCounts.top(); OrderedCounts.pop(); for (map::iterator it = CharCount.begin(); it != CharCount.end(); it++){ if (it->second == currCount){ for (int k = 0; k < currCount; k++){ ans+=it->first; } } } } return ans; } }; {% endblock solution_code %}

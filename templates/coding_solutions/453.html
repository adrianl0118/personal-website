Given a non-empty integer array of size n, find the minimum number of moves required to make all array elements equal, where a move is incrementing n - 1 elements by 1.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int minMoves(vector<int>& nums) { int moves = 0; int min = nums[0]; for (int i = 1; i < nums.size(); i++){ if (nums[i] < min){ min = nums[i]; } } for (int j = 0; j < nums.size(); j++){ moves += (nums[j]-min); } return moves; } }; {% endblock solution_code %}

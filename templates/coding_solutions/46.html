Given a collection of distinct integers, return all possible permutations.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector> permute(vector& nums) { vector> ans; createPerms(ans, 0, nums); return ans; } void createPerms(vector>& ans, int index, vector& nums){ if (index == nums.size()){ ans.push_back(nums); return; } for (int i = index; i < nums.size(); i++){ swap(nums[index],nums[i]); createPerms(ans, index+1, nums); swap(nums[index],nums[i]); } } }; {% endblock solution_code %}

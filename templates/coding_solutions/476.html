Given a positive integer, output its complement number. The complement strategy is to flip the bits of its binary representation.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int findComplement(int num) { vector cnt; while(num > 0){ int cur = num % 2; num /= 2; cnt.push_back(cur); } int res=0; for(int i = 0; i < cnt.size(); i++){ if (cnt[i] == 0){ res += pow(2,i); } } return res; } }; {% endblock solution_code %}

Given a binary array, find the maximum number of consecutive 1s in this array.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int findMaxConsecutiveOnes(vector<int>& nums) { int count = 0; int mcount = -1; for (int i = 0; i < nums.size(); i++){ if (nums[i] == 1){ count++; } else { mcount = max(mcount,count); count = 0; } } return max(mcount,count); } }; {% endblock solution_code %}

You need to find the largest value in each row of a binary tree.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector largestValues(TreeNode* root) { vector LevelMax; DFS(root, 0, LevelMax); return LevelMax; } void DFS (TreeNode* root, int level, vector& LevelMax){ if (!root) { return; } if (LevelMax.size() < level+1){ LevelMax.push_back(root->val); } else { LevelMax[level] = max(LevelMax[level], root->val); } if (root->left){ DFS (root->left, level+1, LevelMax); } if (root->right){ DFS (root->right, level+1, LevelMax); } return; } }; {% endblock solution_code %}

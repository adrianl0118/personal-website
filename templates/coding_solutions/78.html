Given a set of distinct integers, nums, return all possible subsets (the power set).

Note: The solution set must not contain duplicate subsets.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: //power set vector<vector<int>> ans; int n; vector<int> path; set<vector<int>> checker; vector<vector<int>> subsets(vector<int>& nums) { n = nums.size(); dfs(nums , 0, checker); return ans; }; void dfs(const vector<int>& nums, int idx, set<vector<int>>& checker){ if(idx == n){ if (checker.find(path) == checker.end()){ ans.push_back(path); } return; } //current not used dfs(nums , idx + 1, checker); //use current then restore state path.push_back(nums[idx]); dfs(nums , idx + 1, checker); path.pop_back(); return; } }; {% endblock solution_code %}

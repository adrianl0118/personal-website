Given a string S, we can transform every letter individually to be lowercase or uppercase to create another string. Return a list of all possible strings we could create.

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector<string> letterCasePermutation(string S) { vector<string> ans; ans.push_back(S); DownStreamSwap(S, 0, ans); return ans; } //Given a alphabetic char, change its case char SwitchCase(char c){ if (c >= 65 && c <= 90) return c+32; else return c-32; } //Get all downstream swaps from position j void DownStreamSwap(string& S, int j, vector<string>& ans){ for (int i = j; i < S.length(); i++){ if (isalpha(S[i])){ string temp = S; temp[i] = SwitchCase(S[i]); ans.push_back(temp); DownStreamSwap(temp, i+1, ans); } } } }; {% endblock solution_code %}

Given a matrix A, return the transpose of A.

The transpose of a matrix is the matrix flipped over it's main diagonal, switching the row and column indices of the matrix.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector> transpose(vector>& A) { int sidex = A.size(); int sidey = A[0].size(); vector> result; for (int h = 0; h < sidey; h++){ vector temp; result.push_back(temp); } for (int i = 0; i < sidey; i++){ for (int j = 0; j < sidex; j++){ result[i].push_back(A[j][i]); } } return result; } }; {% endblock solution_code %}

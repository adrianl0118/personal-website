We are given two sentences A and B. (A sentence is a string of space separated words. Each word consists only of lowercase letters.)

A word is uncommon if it appears exactly once in one of the sentences, and does not appear in the other sentence.

Return a list of all uncommon words.

You may return the list in any order.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: vector<string> uncommonFromSentences(string A, string B) { //Use a map to count occurrences of words map<string, int> strcount; //Read string A word by word istringstream iss(A); string word; while(iss >> word) { //add the word to the map if not already counted, else count an existing word if (strcount.find(word) == strcount.end()) strcount.insert(pair<string, int>(word,1)); else strcount.at(word)++; } //repeat the process with string A on string B istringstream ist(B); while(ist >> word){ if (strcount.find(word) == strcount.end()) strcount.insert(pair<string, int>(word,1)); else strcount.at(word)++; } //iterating through the map, count words that were only observed once between both strings map<string, int>::iterator it; vector<string> ans;; for (it = strcount.begin(); it != strcount.end(); it++){ if (it->second == 1) ans.push_back(it->first); } return ans; } }; {% endblock solution_code %}

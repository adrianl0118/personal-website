Given an array A of positive lengths, return the largest perimeter of a triangle with non-zero area, formed from 3 of these lengths.

If it is impossible to form any triangle of non-zero area, return 0.

{% endblock question %} {% block solution_exp %}

{% endblock solution_exp %} {% block solution_code %} class Solution { public: int largestPerimeter(vector<int>& A) { int N = A.size(); if (N < 3) { return 0; } // sort A, decreasing sort(A.begin(), A.end(), greater<int>()); // loop from 0 to N-3, only need to consider the combination A[i], A[i+1], A[i+2] for each i for (int i(0); i < N - 2; ++i) { if (A[i] < A[i+1] + A[i+2]) { return A[i] + A[i+1] + A[i+2]; } } return 0; } }; {% endblock solution_code %}
